double wrapAngle(double angle)
{
    while (angle > M_PI) angle -= 2.0 * M_PI;
    while (angle < -M_PI) angle += 2.0 * M_PI;
    return angle;
}

int lowerInterval(const double* values, int count, double value)
{
    if (count < 2) throw std::invalid_argument("Interpolation table must contain at least two values");
    if (value <= values[0]) return 0;
    if (value >= values[count - 1]) return count - 2;
    return static_cast<int>(std::upper_bound(values, values + count, value) - values) - 1;
}

AtmosphereData atmosphere(double h)
{
    if (h > 1500000.0) h = 1500000.0;
    if (h < -1999.0) h = -1999.0;

    double H, T, p, ro, a, g;
    H = round((6356767 * h) / (6356767 + h));
    g = 9.80665 * pow(6356767 / (6356767 + h), 2);

    if (h < 0 && h >= -1999) T = 301.15 - 0.0065 * (H + 2000);
    else if (h < 11019 && h >= 0) T = 288.15 - 0.0065 * H;
    else if (h < 20063 && h >= 11019) T = 216.65;
    else if (h < 32162 && h >= 20063) T = 216.65 + 0.0010 * (H - 20000);
    else if (h < 47350 && h >= 32162) T = 228.65 + 0.0028 * (H - 32000);
    else if (h < 51412 && h >= 47350) T = 270.65;
    else if (h < 71802 && h >= 51412) T = 270.65 - 0.0028 * (H - 51000);
    else if (h < 86152 && h >= 71802) T = 214.65 - 0.0020 * (H - 71000);
    else if (h < 95411 && h >= 86152) T = 186.65;
    else if (h < 104128 && h >= 95411) T = 186.65 + 0.0030 * (H - 94000);
    else if (h < 120000 && h >= 104128) T = 212.00 + 0.0110 * (H - 102450);
    else if (h >= 120000 && h <= 1500000) {
        if (h < 200000) T = 600 + (h - 120000) * (1000 - 600) / 80000;
        else if (h < 500000) T = 1000;
        else if (h < 1000000) T = 1000 + (h - 500000) * (1200 - 1000) / 500000;
        else T = 1200;
    }
    else T = 1200.0;

    a = 20.046796 * sqrt(T);

    if (h < 0 && h >= -1999)
        p = pow(10, (log10(127774) - (9.80665 * log10((301.15 - 0.0065 * (H + 2000)) / 301.15) / (-0.0065 * 287.05287))));
    else if (h < 11019 && h >= 0)
        p = pow(10, (log10(101325) - (9.80665 * log10((288.15 - 0.0065 * H) / 288.15) / (-0.0065 * 287.05287))));
    else if (h < 20063 && h >= 11019)
        p = pow(10, (log10(22632) - (9.80665 * 0.434294 * (H - 11000) / (287.05287 * 216.65))));
    else if (h < 32162 && h >= 20063)
        p = pow(10, (log10(5474.87) - (9.80665 * log10((216.65 + 0.0010 * (H - 20000)) / 216.65) / (0.0010 * 287.05287))));
    else if (h < 47350 && h >= 32162)
        p = pow(10, (log10(868.014) - (9.80665 * log10((228.65 + 0.0028 * (H - 32000)) / 228.65) / (0.0028 * 287.05287))));
    else if (h < 51412 && h >= 47350)
        p = pow(10, (log10(110.906) - (9.80665 * 0.434294 * (H - 47000) / (287.05287 * 270.65))));
    else if (h < 71802 && h >= 51412)
        p = pow(10, (log10(66.9384) - (9.80665 * log10((270.65 - 0.0028 * (H - 51000)) / 270.65) / (-0.0028 * 287.05287))));
    else if (h < 86152 && h >= 71802)
        p = pow(10, (log10(3.95639) - (9.80665 * log10((214.65 - 0.0020 * (H - 71000)) / 214.65) / (-0.0020 * 287.05287))));
    else if (h < 95411 && h >= 86152)
        p = pow(10, (log10(0.363409) - (9.80665 * 0.434294 * (H - 85000) / (287.05287 * 186.65))));
    else if (h < 104128 && h >= 95411)
        p = pow(10, (log10(0.112377) - (9.80665 * log10((186.65 + 0.0030 * (H - 94000)) / 186.65) / (0.0030 * 287.05287))));
    else if (h < 120000 && h >= 104128)
        p = pow(10, (log10(0.0238282) - (9.80665 * log10((212.00 + 0.0110 * (H - 102450)) / 212.00) / (0.0110 * 287.05287))));
    else
    {
        double p_120 = 0.0238282;
        double H_scale = (h < 200000) ? 30000.0 : ((h < 500000) ? 45000.0 : 60000.0);
        p = p_120 * exp(-(h - 120000) / H_scale);
    }
    ro = p / (287.05287 * T);
    return {H, g, T, a, p, ro};
}

// --------------------------------------------------------------
// Тяга (две ступени)
// --------------------------------------------------------------
double Pe(double pressure_pa, double t)
{
    double rash1 = m1 / t1, rash2 = m2 / t2;
    // Замените значения тяги в пустоте на свои
    double P_1 = 276.277 - (276.277 - 247.9424) * ((pressure_pa / 100000.0) / 1.01);
    double P_2 = 278.052;
    if (t <= t1) return P_1 * rash1 * 9.80665;
    if (t <= t1 + t2) return P_2 * rash2 * 9.80665;
    return 0.0;
}

// --------------------------------------------------------------
// Программа тангажа
// --------------------------------------------------------------
double pitchh(const PitchProgram& program, double t, double vx, double vy)
{
    if (t <= program.tv) return M_PI / 2.0;

    if (t <= program.topt) {
        const double fraction = (t - program.tv) / (program.topt - program.tv);
        const double smooth = (1.0 - cos(fraction * M_PI)) / 2.0;
        return M_PI / 2.0 + (program.tettaopt - M_PI / 2.0) * smooth;
    }

    if (t <= tayt) {
        const double fraction = (t - program.topt) / (tayt - program.topt);
        const double smooth = (1.0 - cos(fraction * M_PI)) / 2.0;
        const double target_pitch = program.k1 * tetta(vx, vy);
        return program.tettaopt + (target_pitch - program.tettaopt) * smooth;
    }

    // There is no thrust after burnout. The simplified point-mass model assumes
    // aerodynamic alignment of the separated head section with the velocity vector.
    return tetta(vx, vy);
}

// --------------------------------------------------------------
// Mass and moment of inertia
// --------------------------------------------------------------
double massAt(double t)
{
    t = std::max(0.0, t);

    const double burned_stage1 = m1 * std::clamp(t / t1, 0.0, 1.0);
    const double stage1_propellant = m1 - burned_stage1;
    const double stage1_structure = (t < t1) ? m01 : 0.0;

    const double stage2_elapsed = std::max(0.0, t - t1);
    const double burned_stage2 = m2 * std::clamp(stage2_elapsed / t2, 0.0, 1.0);
    const double stage2_propellant = m2 - burned_stage2;
    const double stage2_structure = (t < tayt) ? m02 : 0.0;

    return mpn + stage1_propellant + stage1_structure + stage2_propellant + stage2_structure;
}
double IZ(double t)
{
    if (t <= t1) return Iz1 + (Iz2 - Iz1) * (t / t1);
    if (t <= t1 + t2) return Iz3 + (Iz4 - Iz3) * ((t - t1) / t2);
    return Iz5;
}

double tetta(double vx, double vy)
{
    double v = sqrt(vx * vx + vy * vy);
    if (v < 1e-3) return M_PI / 2;
    return atan2(vy, vx);
}
double eta(double x, double y) { return atan2(x, R + y); }

// --------------------------------------------------------------
// Point-mass trajectory simulation with RK4 integration
// --------------------------------------------------------------
struct State
{
    double t;
    double x;
    double y;
    double vx;
    double vy;
    double wz;
};

struct Derivative
{
    double x;
    double y;
    double vx;
    double vy;
    double wz;
};

dinamic evaluateState(const State& state, const PitchProgram& program)
{
    dinamic a{};
    a.t = state.t;
    a.x = state.x;
    a.y = state.y;
    a.v_x = state.vx;
    a.v_y = state.vy;
    a.wz = state.wz;
    a.v = hypot(a.v_x, a.v_y);
    a.r = hypot(a.x, R + a.y);
    a.h = a.r - R;
    a.eta = eta(a.x, a.y);
    a.tetta = tetta(a.v_x, a.v_y);
    a.pitch = pitchh(program, a.t, a.v_x, a.v_y);
    a.alpha = wrapAngle(a.pitch - a.tetta);
    a.L = R * a.eta;

    const auto atm = atmosphere(a.h);
    a.M = a.v / atm.a;
    a.q = 0.5 * atm.rho * a.v * a.v;
    a.m = massAt(a.t);
    a.Iz = IZ(a.t);
    a.g = atm.g;
    a.P = Pe(atm.p, a.t);

    const auto aero = Aerodynamic(a.M, a.alpha, a.t, a.q);
    a.X = aero.X;
    a.Y = aero.Y;
    a.Mz = aero.Mz;
    return a;
}

Derivative derivative(const State& state, const PitchProgram& program)
{
    const dinamic a = evaluateState(state, program);
    const double g = a.g;

    Derivative d{};
    d.x = a.v_x;
    d.y = a.v_y;
    d.vx = (a.P * cos(a.pitch) - a.X * cos(a.tetta) - a.Y * sin(a.tetta)
        - a.m * g * sin(a.eta)) / a.m;
    d.vy = (a.P * sin(a.pitch) - a.X * sin(a.tetta) + a.Y * cos(a.tetta)
        - a.m * g * cos(a.eta)) / a.m;
    d.wz = a.Mz / a.Iz;
    return d;
}

State addScaled(const State& state, const Derivative& derivative_value, double factor)
{
    return {
        state.t + factor,
        state.x + derivative_value.x * factor,
        state.y + derivative_value.y * factor,
        state.vx + derivative_value.vx * factor,
        state.vy + derivative_value.vy * factor,
        state.wz + derivative_value.wz * factor
    };
}

State rk4Step(const State& state, const PitchProgram& program, double dt)
{
    const Derivative k1_value = derivative(state, program);
    const Derivative k2_value = derivative(addScaled(state, k1_value, dt / 2.0), program);
    const Derivative k3_value = derivative(addScaled(state, k2_value, dt / 2.0), program);
    const Derivative k4_value = derivative(addScaled(state, k3_value, dt), program);

    State result = state;
    result.t += dt;
    result.x += (k1_value.x + 2.0 * k2_value.x + 2.0 * k3_value.x + k4_value.x) * dt / 6.0;
    result.y += (k1_value.y + 2.0 * k2_value.y + 2.0 * k3_value.y + k4_value.y) * dt / 6.0;
    result.vx += (k1_value.vx + 2.0 * k2_value.vx + 2.0 * k3_value.vx + k4_value.vx) * dt / 6.0;
    result.vy += (k1_value.vy + 2.0 * k2_value.vy + 2.0 * k3_value.vy + k4_value.vy) * dt / 6.0;
    result.wz += (k1_value.wz + 2.0 * k2_value.wz + 2.0 * k3_value.wz + k4_value.wz) * dt / 6.0;
    return result;
}

State interpolateState(const State& left, const State& right, double fraction)
{
    fraction = std::clamp(fraction, 0.0, 1.0);
    return {
        left.t + fraction * (right.t - left.t),
        left.x + fraction * (right.x - left.x),
        left.y + fraction * (right.y - left.y),
        left.vx + fraction * (right.vx - left.vx),
        left.vy + fraction * (right.vy - left.vy),
        left.wz + fraction * (right.wz - left.wz)
    };
}

void writeCsvHeader(std::ofstream& out)
{
    out << "t_s; x_m; y_m; h_m; v_mps; pitch_deg; tetta_deg; alpha_deg; L_m; "
        << "vx_mps; vy_mps; m_kg; M; q_Pa; wz_radps; wz_degps; "
        << "Mz_aero_Nm; Iz_kgm2; wz_dot_radps2; Cx; Cy; mz; X_N; Y_N\n";
}

void writeCsvRow(std::ofstream& out, const dinamic& a)
{
    const auto aero = Aerodynamic(a.M, a.alpha, a.t, a.q);
    const double wz_dot = aero.Mz / a.Iz;
    out << a.t << "; " << a.x << "; " << a.y << "; " << a.h << "; "
        << a.v << "; " << radToDeg(a.pitch) << "; " << radToDeg(a.tetta) << "; "
        << radToDeg(a.alpha) << "; " << a.L << "; " << a.v_x << "; " << a.v_y << "; "
        << a.m << "; " << a.M << "; " << a.q << "; " << a.wz << "; " << radToDeg(a.wz) << "; "
        << aero.Mz << "; " << a.Iz << "; " << wz_dot << "; "
        << aero.Cx << "; " << aero.Cy << "; " << aero.mz << "; "
        << aero.X << "; " << aero.Y << "\n";
}

bool validProgram(const PitchProgram& program)
{
    return program.tv > 8.0
        && program.tv < program.topt
        && program.topt < tayt
        && program.topt - program.tv >= MIN_TOPT_GAP
        && tayt - program.topt >= MIN_TOPT_GAP
        && std::isfinite(program.k1)
        && std::isfinite(program.tettaopt);
}

SimulationResult simulate(const PitchProgram& program, double dt, bool write_csv, const std::string& csv_path)
{
    SimulationResult result{};
    if (!validProgram(program) || !(dt > 0.0)) return result;

    std::ofstream out;
    if (write_csv) {
        out.open(csv_path, std::ios::out);
        if (!out) throw std::runtime_error("Cannot open output file: " + csv_path);
        out << std::fixed << std::setprecision(8);
        writeCsvHeader(out);
    }

    State state{0.0, 0.0, 30.0, 0.0, 10.0, 0.0};
    dinamic current = evaluateState(state, program);
    result.max_height_m = current.h;
    result.max_abs_alpha_deg = std::abs(radToDeg(current.alpha));
    if (write_csv) writeCsvRow(out, current);

    constexpr double MAX_SIMULATION_TIME_S = 3500.0;
    while (state.t < MAX_SIMULATION_TIME_S) {
        double step = dt;

        // Land exactly on stage events to keep discontinuities out of RK4 substeps.
        for (const double event_time : {t1, tayt}) {
            if (state.t < event_time && state.t + step > event_time) {
                step = event_time - state.t;
            }
        }

        const State previous_state = state;
        const dinamic previous = current;
        state = rk4Step(state, program, step);
        current = evaluateState(state, program);

        if (!std::isfinite(current.h) || !std::isfinite(current.L) || !std::isfinite(current.v)
            || !std::isfinite(current.alpha) || current.m <= 0.0) {
            return result;
        }

        if (current.h < 0.0) {
            const double fraction = previous.h / (previous.h - current.h);
            const State impact_state = interpolateState(previous_state, state, fraction);
            current = evaluateState(impact_state, program);
        }

        result.max_height_m = std::max(result.max_height_m, current.h);
        result.max_abs_alpha_deg = std::max(result.max_abs_alpha_deg, std::abs(radToDeg(current.alpha)));
        if (write_csv) writeCsvRow(out, current);

        if (current.h <= 0.0 && state.t > 0.0) {
            result.valid = true;
            result.range_m = current.L;
            result.impact_time_s = current.t;
            result.impact_speed_mps = current.v;
            return result;
        }
    }

    return result;
}

// --------------------------------------------------------------
// Differential-evolution optimizer
// --------------------------------------------------------------
using EncodedProgram = std::array<double, 4>;

struct Candidate
{
    EncodedProgram x{};
    PitchProgram program{};
    SimulationResult result{};
};

PitchProgram decodeProgram(const EncodedProgram& encoded)
{
    const auto map = [](double value, double lower, double upper) {
        value = std::clamp(value, 0.0, 1.0);
        return lower + value * (upper - lower);
    };

    PitchProgram program{};
    program.tv = map(encoded[0], MIN_TV, MAX_TV);
    const double min_topt = program.tv + MIN_TOPT_GAP;
    const double max_topt = tayt - MIN_TOPT_GAP;
    program.topt = map(encoded[1], min_topt, max_topt);
    program.k1 = map(encoded[2], MIN_K1, MAX_K1);
    program.tettaopt = degToRad(map(encoded[3], MIN_TETTAOPT_DEG, MAX_TETTAOPT_DEG));
    return program;
}

double constraintViolation(const Candidate& candidate)
{
    if (!candidate.result.valid) return 1.0e9;
    const double alpha_excess = std::max(0.0, candidate.result.max_abs_alpha_deg - ALPHA_LIMIT_DEG);
    const double range_shortfall = std::max(0.0, MIN_RANGE_KM - candidate.result.range_m / 1000.0);
    return 10.0 * alpha_excess / ALPHA_LIMIT_DEG + range_shortfall / MIN_RANGE_KM;
}

bool isFeasible(const Candidate& candidate)
{
    return candidate.result.valid
        && candidate.result.max_abs_alpha_deg < ALPHA_LIMIT_DEG
        && candidate.result.range_m / 1000.0 >= MIN_RANGE_KM;
}

bool betterCandidate(const Candidate& left, const Candidate& right)
{
    const bool left_feasible = isFeasible(left);
    const bool right_feasible = isFeasible(right);
    if (left_feasible != right_feasible) return left_feasible;

    if (left_feasible) return left.result.range_m > right.result.range_m;

    const double left_violation = constraintViolation(left);
    const double right_violation = constraintViolation(right);
    if (std::abs(left_violation - right_violation) > 1.0e-12) return left_violation < right_violation;
    if (left.result.max_abs_alpha_deg != right.result.max_abs_alpha_deg) {
        return left.result.max_abs_alpha_deg < right.result.max_abs_alpha_deg;
    }
    return left.result.range_m > right.result.range_m;
}

Candidate evaluateCandidate(const EncodedProgram& encoded, double dt)
{
    Candidate candidate{};
    candidate.x = encoded;
    candidate.program = decodeProgram(encoded);
    candidate.result = simulate(candidate.program, dt, false);
    return candidate;
}

EncodedProgram encodeProgram(const PitchProgram& program)
{
    const auto invMap = [](double value, double lower, double upper) {
        return std::clamp((value - lower) / (upper - lower), 0.0, 1.0);
    };
    EncodedProgram encoded{};
    encoded[0] = invMap(program.tv, MIN_TV, MAX_TV);
    const double min_topt = program.tv + MIN_TOPT_GAP;
    const double max_topt = tayt - MIN_TOPT_GAP;
    encoded[1] = invMap(program.topt, min_topt, max_topt);
    encoded[2] = invMap(program.k1, MIN_K1, MAX_K1);
    encoded[3] = invMap(radToDeg(program.tettaopt), MIN_TETTAOPT_DEG, MAX_TETTAOPT_DEG);
    return encoded;
}

void printCandidate(const Candidate& candidate, const std::string& prefix)
{
    std::cout << prefix
        << " tv=" << candidate.program.tv
        << " s, topt=" << candidate.program.topt
        << " s, k1=" << candidate.program.k1
        << ", tettaopt=" << radToDeg(candidate.program.tettaopt)
        << " deg | L=" << candidate.result.range_m / 1000.0
        << " km, max|alpha|=" << candidate.result.max_abs_alpha_deg
        << " deg, feasible=" << (isFeasible(candidate) ? "yes" : "no") << "\n";
}

Candidate optimizePitchProgram(int population_size, int generations, double optimization_dt, std::uint64_t seed)
{
    if (population_size < 8) throw std::invalid_argument("Population size must be at least 8");
    if (generations < 1) throw std::invalid_argument("Generation count must be positive");

    std::mt19937_64 rng(seed);
    std::uniform_real_distribution<double> uniform(0.0, 1.0);
    std::uniform_int_distribution<int> pick(0, population_size - 1);

    std::vector<Candidate> population;
    population.reserve(population_size);

    // Keep a physically reasonable seed near the original program, but satisfy tv > 8 s.
    const PitchProgram seed_program{9.0, 35.0, 0.98, degToRad(43.0)};
    population.push_back(evaluateCandidate(encodeProgram(seed_program), optimization_dt));
    while (static_cast<int>(population.size()) < population_size) {
        EncodedProgram encoded{};
        for (double& value : encoded) value = uniform(rng);
        population.push_back(evaluateCandidate(encoded, optimization_dt));
    }

    auto best = [&]() -> Candidate {
        return *std::min_element(population.begin(), population.end(), [](const Candidate& a, const Candidate& b) {
            return betterCandidate(a, b);
        });
    };

    constexpr double differential_weight = 0.78;
    constexpr double crossover_probability = 0.90;

    for (int generation = 1; generation <= generations; ++generation) {
        for (int i = 0; i < population_size; ++i) {
            int a, b, c;
            do { a = pick(rng); } while (a == i);
            do { b = pick(rng); } while (b == i || b == a);
            do { c = pick(rng); } while (c == i || c == a || c == b);

            EncodedProgram trial = population[i].x;
            const int forced_dimension = static_cast<int>(rng() % trial.size());
            for (int j = 0; j < static_cast<int>(trial.size()); ++j) {
                if (uniform(rng) <= crossover_probability || j == forced_dimension) {
                    trial[j] = population[a].x[j] + differential_weight * (population[b].x[j] - population[c].x[j]);
                    trial[j] = std::clamp(trial[j], 0.0, 1.0);
                }
            }

            Candidate evaluated = evaluateCandidate(trial, optimization_dt);
            if (betterCandidate(evaluated, population[i])) population[i] = evaluated;
        }

        if (generation == 1 || generation % 10 == 0 || generation == generations) {
            printCandidate(best(), "Generation " + std::to_string(generation) + ":");
        }
    }

    // Re-rank the final coarse population with the final integration step.
    Candidate best_fine = evaluateCandidate(population.front().x, 0.1);
    for (const Candidate& coarse : population) {
        Candidate fine = evaluateCandidate(coarse.x, 0.1);
        if (betterCandidate(fine, best_fine)) best_fine = fine;
    }

    // Small deterministic local random refinement around the best point.
    EncodedProgram center = best_fine.x;
    for (int round = 0; round < 4; ++round) {
        const double radius = 0.08 * std::pow(0.45, round);
        std::uniform_real_distribution<double> perturb(-radius, radius);
        for (int attempt = 0; attempt < 48; ++attempt) {
            EncodedProgram trial = center;
            for (double& value : trial) value = std::clamp(value + perturb(rng), 0.0, 1.0);
            Candidate fine = evaluateCandidate(trial, 0.1);
            if (betterCandidate(fine, best_fine)) {
                best_fine = fine;
                center = best_fine.x;
            }
        }
    }

    return best_fine;
}

void printUsage(const char* executable)
{
    std::cout << "Usage: " << executable
        << " [--population N] [--generations N] [--opt-dt seconds] [--final-dt seconds] [--seed N]\n";
}

int main(int argc, char** argv)
{
    setlocale(LC_ALL, "Russian");

    int population_size = 32;
    int generations = 36;
    double optimization_dt = 0.40;
    double final_dt = 0.1;
    std::uint64_t seed = 20260531ULL;

    for (int i = 1; i < argc; ++i) {
        const std::string arg = argv[i];
        auto requireValue = [&](const std::string& option) -> std::string {
            if (i + 1 >= argc) throw std::invalid_argument("Missing value for " + option);
            return argv[++i];
        };

        if (arg == "--population") population_size = std::stoi(requireValue(arg));
        else if (arg == "--generations") generations = std::stoi(requireValue(arg));
        else if (arg == "--opt-dt") optimization_dt = std::stod(requireValue(arg));
        else if (arg == "--final-dt") final_dt = std::stod(requireValue(arg));
        else if (arg == "--seed") seed = std::stoull(requireValue(arg));
        else if (arg == "--help") { printUsage(argv[0]); return 0; }
        else throw std::invalid_argument("Unknown argument: " + arg);
    }

    std::cout << std::fixed << std::setprecision(6);
    std::cout << "Optimizing pitch program for L >= " << MIN_RANGE_KM
        << " km and |alpha| < " << ALPHA_LIMIT_DEG << " deg\n";

    Candidate best = optimizePitchProgram(population_size, generations, optimization_dt, seed);
    best.result = simulate(best.program, final_dt, true, "trajectory_optimized.csv");

    std::cout << "\nFinal validated solution (dt=" << final_dt << " s):\n";
    printCandidate(best, "Best:");
    std::cout << "Maximum height = " << best.result.max_height_m / 1000.0 << " km\n";
    std::cout << "Impact time = " << best.result.impact_time_s << " s\n";
    std::cout << "Impact speed = " << best.result.impact_speed_mps << " m/s\n";
    std::cout << "Trajectory saved to trajectory_optimized.csv\n";

    if (!isFeasible(best)) {
        std::cerr << "No feasible program found within the configured search budget. "
            << "Increase --population and/or --generations.\n";
        return 2;
    }
    return 0;
}
